@Component
public class CryptoAndDecrypt {

    @Value("${keys.clientPublicKey}")
    private String clientPublicKey;

    @Value("${keys.clientPrivateKey}")
    private String clientPrivateKey;

    @Value("${keys.serverPublicKey}")
    private String serverPublicKey;

    @Value("${keys.serverPrivateKey}")
    private String serverPrivateKey;

    /**
     * @param encryptkey
     * @param postData
     * @Title: decrypt
     * @Description: 解密
     * @author: viking
     * @date: 2018/11/6 上午10:52
     * @return: com.alibaba.fastjson.JSONObject
     * @throw:
     */
    public JSONObject decrypt(String encryptkey, String postData) throws Exception {
        // 验签
        boolean passSign = EncryUtil.checkDecryptAndSign(postData,
                encryptkey, clientPublicKey, serverPrivateKey);

        if (passSign) {
            // 验签通过
            String aeskey = RSA.decrypt(encryptkey,
                    serverPrivateKey);
            String data = AES.decryptFromBase64(postData,
                    aeskey);

            JSONObject jsonObj = JSONObject.parseObject(data);
            return jsonObj;
        }
        return null;
    }

    /**
     * @param params
     * @Title: crypto
     * @Description: 加密
     * @author: viking
     * @date: 2018/11/6 上午11:00
     * @return: java.lang.String
     * @throw:
     */
    public String crypto(TreeMap<String, Object> params) throws Exception {
        // 生成RSA签名
        String sign = EncryUtil.handleRSA(params, clientPrivateKey);
        params.put("sign", sign);

        String info = JSON.toJSONString(params);
        //随机生成AES密钥
        String aesKey = RandomUtil.getRandom(16);
        //AES加密数据
        String data = AES.encryptToBase64(info, aesKey);

        // 使用RSA算法将商户自己随机生成的AESkey加密
        String encryptkey = RSA.encrypt(aesKey, serverPublicKey);

        Map<String, Object> map = new HashMap<>();
        map.put("data", data);
        map.put("encryptkey", encryptkey);
        return JSON.toJSONString(map);
    }
}
