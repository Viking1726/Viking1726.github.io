# 服务器上nginx和rmtp模块安装部分

下载nginx和rtmp模块

nginx-rtmp-module github主页

wget http://nginx.org/download/nginx-1.15.5.tar.gz
wget https://github.com/arut/nginx-rtmp-module/archive/master.zip

zsh

zsh extract 插件 解压直接 x 文件名.任意类型

apt-get install build-essential libpcre3 libpcre3-dev libssl-dev //这里可以先不执行,等报错了在安装
cd nginx-1.15.5
./configure --with-http_ssl_module --add-module=../master/nginx-rtmp-module-master
make
make instasll
wget https://raw.github.com/JasonGiedymin/nginx-init-ubuntu/master/nginx -O /etc/init.d/nginx
chmod +x /etc/init.d/nginx
service nginx start

nginx可能默认加入的模块

若有报错加入--without-http_gzip_module，或者考虑下载该模块并加入。

./configure --with-http_ssl_module --add-module=../master/nginx-rtmp-module-master --without-http_gzip_module

# nginx.conf 部分

记得替换ip

推流路径 rtmp://your_server_ip:1935/hls
拉流路径 rtmp://your_server_ip:1935/hls

http {
        ...

    server {
        listen       80;
        server_name  localhost;

        ...

}
rtmp {
    server {
        listen 1935;
        application hls {
            live on;
            record off;
        }
    }
}

# nginx.conf 部分二

记得替换ip

推流路径 rtmp://YOUR_SERVER_IP:1935/hls
拉流路径 http://YOUR_SERVER_IP/hls/index.m3u8

http {
        ...

    server {
        listen       80;
        server_name  localhost;

        ...
        
    location /hls {  
                types {  
                    application/vnd.apple.mpegurl m3u8;  
                    video/mp2t ts;  
                }  
                root /tmp;
                #alias /tmp/hls;    # root 和 alias任选一种。
                expires -1;         #指定“Expires”的值为当前服务器时间-1s，即永远过期。 
                Cache-Control no-cache;  
    }  


}
rtmp {
    server {
        listen 1935;
        application hls {
            live on;
            hls on;                  
            wait_key on;             #对视频切片进行保护，这样就不会产生马赛克了。
            hls_path /tmp/hls;       #切片存放位置。
            hls_fragment 10s;        #切片的时长。
            hls_playlist_length 60s; #总时长。
            hls_continuous on;       #连续模式。
            hls_cleanup on;          #切片清理。
            hls_nested on;           #嵌套模式。
        }
    }
}
