# demo

import cv2
cap = cv2.VideoCapture("rtmp://YOUR_SERVER_IP:1935/hls")
ret,frame = cap.read()
while ret:
    ret,frame = cap.read()
    cv2.imshow("frame",frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cv2.destroyAllWindows()
cap.release()
