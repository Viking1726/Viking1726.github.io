    function addImage(val) {
        var imgFlag = $("#imgGroup").find("img").length;
        if (val) {
            imgId = val;
        }
        if (imgFlag <= 5) {//控制上传数量
            $("#inputImg").click();
        } else {
            return;
        }
    }

    //图片压缩
    function imgResize(file, callback) {
        var fileReader = new FileReader();
        fileReader.onload = function () {
            var IMG = new Image();
            IMG.src = this.result;
            IMG.onload = function () {
                var w = this.naturalWidth, 
                    h = this.naturalHeight, 
                    resizeW = 0, 
                    resizeH = 0;
                // maxSize 是压缩的设置，设置图片的最大宽度和最大高度，等比缩放，level是报错的质量，数值越小质量越低
                var maxSize = {
                    width: 500,
                    height: 500,
                    level: 0.6
                };
                if (w > maxSize.width || h > maxSize.height) {
                    var multiple = Math.max(w / maxSize.width, h / maxSize.height);
                    resizeW = w / multiple;
                    resizeH = w / multiple;
                } else {
                    // 如果图片尺寸小于最大限制，则不压缩直接上传
                    return callback(file)
                }
                var canvas = document.createElement('canvas'),
                    ctx = canvas.getContext('2d');
                if (window.navigator.userAgent.indexOf('iPhone') > 0) {
                    canvas.width = resizeH;
                    canvas.height = resizeW;
                    ctx.rorate(90 * Math.PI / 180);
                    ctx.drawImage(IMG, 0, -resizeH, resizeW, resizeH);
                } else {
                    canvas.width = resizeW;
                    canvas.height = resizeH;
                    ctx.drawImage(IMG, 0, 0, resizeW, resizeH);
                }
                var base64 = canvas.toDataURL('image/jpeg', maxSize.level);
                convertBlob(window.atob(base64.split(',')[1]), callback);
            }
        };
        fileReader.readAsDataURL(file);
    }

    //base64 转 blob
    function convertBlob(base64, callback) {
        var buffer = new ArrayBuffer(base64.length);
        var ubuffer = new Uint8Array(buffer);
        for (var i = 0; i < base64.length; i++) {
            ubuffer[i] = base64.charCodeAt(i)
        }
        var blob;
        try {
            blob = new Blob([buffer], {type: 'image/jpg'});
        } catch (e) {
            window.BlobBuilder = window.BlobBuilder || window.WebKitBlobBuilder || window.MozBlobBuilder || window.MSBlobBuilder;
            if (e.name === 'TypeError' && window.BlobBuilder) {
                var blobBuilder = new BlobBuilder();
                blobBuilder.append(buffer);
                blob = blobBuilder.getBlob('image/jpg');
            }
        }
        callback(blob)
    }

    // 后台保存图片
    function uploadImg(file) {
        $.ajaxSetup({
            data: {csrftoken: window.csrftoken},
        });
        var formData = new FormData();
        formData.append("files", file);
        $.ajax({
            type: "POST",
            url: "/fileUpload",
            data: formData,
            processData: false,
            contentType: false,
            success: function (result) {
                result = JSON.parse(result)
                if (result.status == "success") {
                    showImg(result.url, result.subUrl);
                }
            },
            error: function (result) {
                console.log(result)
            }
        });
    }

    //页面展示图片
    function showImg(url, subUrl) {
        var imgFlag = $("#imgGroup").find("img").length;
        if (imgId) {
            var html = '<img src="' + url + '" style="max-width: 100%;max-height:100% ">';
            $("#" + imgId).html(html);
            var index = imgId.split("-")[1];
            var _url = $("#recordImgUrl").val();
            var urlList = _url.split(",");
            urlList.splice(index - 1, 1, subUrl);
            $("#recordImgUrl").val(urlList.join(","));
        } else {
            var uuid = "uuid-" + imgFlag;
            var html =
                '<div class="form-group col-sm-3">\n' +
                '    <div class="addImg" id="' + uuid + '" style="width:100%;height:122px;text-align: center;border: 1px solid #ddd;-webkit-border-radius: 3px;-moz-border-radius: 3px;border-radius: 3px;" onclick="addImage(\'' + uuid + '\');">\n' +
                '        <img src="' + url + '" style="max-width: 100%;max-height:100% ">' +
                '    </div>\n' +
                '</div>';
            $("#addImg").before(html);
            var _url = $("#recordImgUrl").val();
            if (_url) {
                $("#recordImgUrl").val(_url + "," + subUrl);
            } else {
                $("#recordImgUrl").val(subUrl);
            }
        }
        if (imgFlag == 5) {
            $("#addImg").attr("style", "display: none")
            $("#addImg2").attr("style", "display: none")
        }
    }
