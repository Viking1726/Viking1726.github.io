# MYSQL 片段

# 根据条件去除数据库重复数据

delete
from "table"
where "column" in
      (select t."column" from (
      select "column" from "table" group by "column" having count(*) > 1) t)

# 替换字段特殊字符

update "table"
set "column" = replace("column", '/', '');

特殊符号

char(10): 换行符
char(13): 回车键

# 截取合并

update "table"
set "column" = concat(substring_index("column", '/', 3), substring_index("column", '/', -1));

//concat 合并结果 substring_index 分割截取
//若"column"的值为  aaaa/bbbb/cccc/dddd  
//substring_index("column", '/', 3) = aaaa/bbbb/cccc 
//substring_index("column", '/', -1) = dddd 
//concat(substring_index("column", '/', 3), substring_index("column", '/', -1)) = aaaa/bbbb/ccccdddd

# locate

LOCATE(substr,str) 
//返回子串 substr 在字符串 str 中第一次出现的位置。如果子串 substr 在 str 中不存在，返回值为 0
LOCATE(substr,str,pos) 
//返回子串 substr 在字符串 str 中的第 pos 位置后第一次出现的位置。如果 substr 不在 str 中返回 0

# 使用case-when 排序

select * from t_msg 
order by case when locate(substr, str) > 0 then 1 else 0 end  // desc | asc
